    <script>

        async function cargarContenido() {

            const container = document.getElementById('cards-container');

            

            try {

                const response = await fetch('contenido.txt');

                const textoCompleto = await response.text();


                // Separamos por las líneas "---"

                const bloques = textoCompleto.split('---');


                container.innerHTML = '';


                bloques.forEach(bloque => {

                    if(bloque.trim() === "") return;


                    // Extraer campos usando expresiones regulares simples

                    const titulo = extraerCampo(bloque, "Título");

                    const descripcion = extraerCampo(bloque, "Descripción");

                    const materiales = extraerCampo(bloque, "Materiales");


                    const slide = document.createElement('div');

                    slide.className = 'swiper-slide';

                    

                    slide.innerHTML = `

                        <div class="card">

                            <h1 class="title">${titulo}</h1>

                            <p class="description">${descripcion}</p>

                            <div class="materials-container">

                                <div class="materials-title">Materiales necesarios:</div>

                                <div class="materials-text">${materiales}</div>

                            </div>

                        </div>

                    `;

                    container.appendChild(slide);

                });


                initSwiper();


            } catch (error) {

                console.error(error);

                container.innerHTML = '<div class="swiper-slide">Error al cargar contenido.txt</div>';

            }

        }


        // Función auxiliar para buscar el texto después de "Campo:"

        function extraerCampo(texto, campo) {

            const regex = new RegExp(`${campo}:\\s*(.*)`, "i");

            const match = texto.match(regex);

            return match ? match[1].trim() : "";

        }


        function initSwiper() {

            new Swiper('.swiper', {

                direction: 'vertical',

                slidesPerView: 1,

                mousewheel: true,

                resistanceRatio: 0.4,

            });

        }


        cargarContenido();

    </script>
